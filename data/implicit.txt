
layout(local_size_x=1, local_size_y=1, local_size_z=1) in;

struct sphere_instance
{
	vec3 color;
	float radius;
};


struct cylinder_instance
{
	vec3 color;

	/* radius of each cylinder  */
	float radius; 

	/* height of cylinder */
	float height;
	
	/* Required padding */
	float dummy[3];
};


struct box_instance
{	
	vec3 color; 
	
	float padding1;

	/* box dimension */	
	vec3 dimension;

	float padding2;
};




#define	UNION 	0



struct object_instance
{
	uint levels;
	uint index_type;
	uint index_left;
	uint index_right;
	
	mat4 translation;
	mat4 translation_inv;
};


struct object_node
{
	uint padding;
	uint index_type;
	uint object_index;
	uint node_index;
	
	mat4 translation;
	mat4 translation_inv;
};

float sphere(vec3 p, float r)
{
	return length(p) - r;
}


float cylinder(vec3 p, float height, float radius)
{
	vec2 d = abs(vec2(length(p.xz), p.y)) - vec2(radius, height);
	return min(max(d.x, d.y), 0.0) + length(max(d, 0.0));			
}

float box(vec3 p, vec3 d){			
	vec3 q = abs(p) - d;				
	return length(max(q, 0.0)) + min(max(q.x, max(q.y, q.z)), 0.0);	
}	

struct distance_result
{
	vec4 p;
	float min;
	vec3 color;
};


distance_result distance_object(vec4 p, float t_max, uint index)
{
	
	vec4 p_res = vec4(0.0);
	vec3 color = vec3(0.0);
	float min_t = t_max;
	
	if(index < spheres_count)
	{
		float t = sphere(p.xyz, spheres[index].radius);
		color = spheres[index].color;
		p_res = p;
		min_t = t;
			
	}
	else if(index - spheres_count < cylinders_count)
	{
		uint i = index - spheres_count;
		vec4 e = p;
		float t = cylinder(p.xyz, cylinders[i].radius, cylinders[i].height);
	
		color = cylinders[i].color;
		p_res = p;
		min_t = t;
		
	}
	else if(index - spheres_count - cylinders_count < boxes_count)
	{
		uint i = index - spheres_count - cylinders_count;
		vec4 e = p;
		float t = box(p.xyz, boxes[i].dimension);
	
		color = boxes[i].color;
		p_res = p;
		min_t = t;
		
	} 
	
	distance_result result;
	result.p = p_res;
	result.min = min_t;
	result.color = color;
	return result;
	
}

distance_result distance(vec4 p, float t_max)
{
	
	distance_result result;
	result.min = t_max;
	
	distance_result result_object_none;
	result_object_none.min = t_max;
	result_object_none.color = vec3(1.0);
	
	
	for(int i = 0; i < objects_count; i++)
	{
		uint left_nodes[OBJECT_NODE_STACK];
		uint right_nodes[OBJECT_NODE_STACK];
		
		left_nodes[0] = objects[i].index_left;	
		right_nodes[0] = objects[i].index_right;	
		
		mat4 left_translation_inv = objects[i].translation_inv * nodes[objects[i].index_left].translation_inv;
		mat4 right_translation_inv = objects[i].translation_inv * nodes[objects[i].index_right].translation_inv;
		
		/* Write node index to buffer */
		for(uint i = 1; i < objects[i].levels; i++)
		{
			left_nodes[i] = nodes[left_nodes[i-1]].node_index;
			right_nodes[i] = nodes[right_nodes[i-1]].node_index;
			
			left_translation_inv = left_translation_inv * nodes[left_nodes[i]].translation_inv;
			right_translation_inv = right_translation_inv * nodes[right_nodes[i]].translation_inv;
		}
		
		distance_result left_result[OBJECT_NODE_STACK];
		distance_result right_result[OBJECT_NODE_STACK];

		/* Access the nodes backwards */
		uint index = objects[i].levels-1;
		
		uint left = left_nodes[index];	
		uint right = right_nodes[index];	

		left_result[index] = result_object_none;
		switch(nodes[left].index_type)
		{
			case UNION:
			{
				distance_result left_res = distance_object(left_translation_inv * p, t_max, nodes[left].object_index);
				left_result[index] = left_res;
			}
			break;
		}
		
		left_result[index].p = nodes[left].translation * left_result[index].p; 
		left_translation_inv = left_translation_inv * nodes[left].translation;

		right_result[index] = result_object_none;
		switch(nodes[left].index_type)
		{
			case UNION:
			{
				distance_result right_res = distance_object(right_translation_inv * p, t_max, nodes[right].object_index);
				right_result[index] = right_res;
			}
			break;
		}
		right_result[index].p = nodes[right].translation * right_result[index].p; 
		right_translation_inv = right_translation_inv * nodes[right].translation;


		for(uint i = 2; i < objects[i].levels+1; i++)
		{
			uint index = objects[i].levels - i;
			uint left = left_nodes[index];	
			uint right = right_nodes[index];	
			
			left_result[index] = result_object_none;
			switch(nodes[left].index_type)
			{
				case UNION:
				{
					distance_result left_res = distance_object(left_translation_inv * p, t_max, nodes[left].object_index);
					left_result[index] = left_result[index+1].min < left_res.min ? left_result[index+1] : left_res; 
				}
				break;
			}
			left_result[index].p = nodes[left].translation * left_result[index].p; 
			left_translation_inv = left_translation_inv * nodes[left].translation;


			right_result[index] = result_object_none;
			switch(nodes[left].index_type)
			{
				case UNION:
				{
					distance_result right_res = distance_object(right_translation_inv * p, t_max, nodes[right].object_index);
					right_result[index] = right_result[index+1].min < right_res.min ? right_result[index+1] : right_res; 
				}
				break;
			}
			right_result[index].p = nodes[right].translation * right_result[index].p; 
			right_translation_inv = right_translation_inv * nodes[right].translation;
			
		}

		distance_result res = left_result[0].min < right_result[0].min ? left_result[0] : right_result[0];
		res.p = objects[i].translation * res.p;
		
		result = res.min < result.min ? res : result;
	} 
	
	
	return result;
	
}

void main()
{
	ivec2 coord = ivec2(gl_GlobalInvocationID.xy);
	vec2 uv = 2.0*vec2(gl_GlobalInvocationID.xy)/gl_NumWorkGroups.xy-1.0;
	uv.x *= float(gl_NumWorkGroups.x)/float(gl_NumWorkGroups.y);
	vec3 ray = normalize(vec3(uv, 1.0));
	
	
	imageStore(normal, coord, vec4(0.0, 0.0, 0.0, 1.0));
	imageStore(color, coord, vec4(0.0, 0.0, 0.0, 1.0));
	imageStore(position, coord, vec4(0.0, 0.0, 0.0, 0.0));
			
	float t_max = 100.0;
	float t = 0.0; //imageLoad(buffer_t, coord/4).x;
	
	distance_result d_res; 
	for(int i = 0; i < 128; i++)
	{
		d_res = distance(view_inv*vec4(ray*t, 1.0), t_max);
		if(d_res.min < 0.01 || t > t_max )
		{
			break;
		}
		t = t + d_res.min;
	}
	
	if(t < t_max)
	{
		vec4 p = view*d_res.p;
		
		vec3 e = vec3(0.01, 0.0, 0.0);
		
		vec4 n_ray = view_inv*vec4(ray*t, 1.0);
		vec3 n = vec3(
			distance(vec4(n_ray+e.xyyy), t_max).min - distance(vec4(n_ray-e.xyyy), t_max).min,
			distance(vec4(n_ray+e.yxyy), t_max).min - distance(vec4(n_ray-e.yxyy), t_max).min,
			distance(vec4(n_ray+e.yyxy), t_max).min - distance(vec4(n_ray-e.yyxy), t_max).min
		);
		//n = (view_inv*vec4(n, 1.0)).xyz;	
		n = normalize(-n);	
		
		imageStore(color, coord, vec4(d_res.color, 1.0));
		imageStore(position, coord, vec4(p.xyz, 0.0));
		imageStore(normal, coord, vec4(n, 0.0));
	}
}
